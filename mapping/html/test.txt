    var func_searchbox = function() {
        
        $('#inputsearchall_full').on('keypress', function (event) {

            if(event.which === 13){
                $('#search_box_full').submit();
            }
        });
        var cache = {};
        $( "#inputsearchall_full" ).autocomplete({
            minLength: 2,
            appendTo: '#cols-right',
            source: function( request, response ) {
                var term = request.term;
                var gate = 4;
                term = term + ' / ' + gate;
                if ( term in cache ) {
                    response( cache[ term ] );
                    return;
                }
                request.gate = gate;

                $.getJSON( "/api/searchTtd", request, function( data, status, xhr ) {
                    cache[ term ] = data;
                    response( data );
                });
            },
            select: function( event, ui ) {
                $(this).val(ui.item.label);
                $('#search_box_full').submit();
                return false;
            },
            open: function() { $('.ui-menu').width($(this).width() + 20) }
        });

            dim_color_other_element($('#btn_search_time'));
            dim_color_other_element($('#btn_search_time3'));
            }, 1000
        );
    }

    if( document.addEventListener ) {
        addListener(document, "DOMContentLoaded", func_searchbox);
    }else{
        addListener(window, "load", func_searchbox);
